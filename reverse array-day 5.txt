#include <stdio.h> 
int main() {
    int arr[100], n, i;
    printf("Enter the number of elements: ");
    scanf("%d", &n);
    printf("Enter %d elements:\n", n);
    for (i = 0; i < n; i++){
        scanf("%d", &arr[i]);
        }
        printf("Original array: ");
        for (i = 0; i < n; i++) {
            printf("%d ", arr[i]);
            }
            printf("\n");
            int start = 0; int end = n - 1; while (start < end) {
                int temp = arr[start];
                arr[start] = arr[end];
                arr[end] = temp; 
                start++; end--;
                }
                printf("Reversed array: ");
                for (i = 0; i < n; i++) { 
                    printf("%d ", arr[i]);
                    } printf("\n");
                    return 0;
                    }
